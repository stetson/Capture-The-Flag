 26 /**
 27  * Purge old players
 28  * 
 29  * @memberOf players
 30  * @name purge_players
 31  */
 32 purge_players = function() {
 33 	for (player_iterator in players) {
 34 		// Purge players who haven't updated in over 15 seconds
 35 		if (new Date() - players[player_iterator].last_update > 15000) {
 36 			players[player_iterator].latitude = 0;
 37 			players[player_iterator].longitude = 0;
 38 			players[player_iterator].accuracy = 0;
 39 		}
 40 		
 41 		// Reclaim memory of players who haven't updated in 5 minutes
 42 		if (new Date() - players[player_iterator].last_update > 300000) {
 43 			delete players[player_iterator];
 44 		}
 45 	}
 46 };
 47 
 48 setInterval(purge_players, 5000);

 56 /**
 57  * Update the user's location, and fetch the locations of
 58  * the other players
 59  * 
 60  * @memberOf views
 61  * @name update_location
 62  * @param id {Number} The user's id
 63  * @param latitude {Number}	The user's current latitude
 64  * @param longitude {Number} The user's current longitude
 65  * @param accuracy {Number} The accuracy of the location in meters
 66  */
 67 exports.update_location = function(request, response) {	
 68 	// Record user's location
 69 	try {
 70 		players[request.body.user_id] = {
 71 			'latitude': request.body.latitude,
 72 			'longitude': request.body.longitude,
 73 			'accuracy': request.body.accuracy,
 74 			'last_update': new Date(),
 75 			'auth_token': request.body.auth_token
 76 		};
 77 	} catch (e) { }
 78 	
 79 	// Send the players back to the client
 80 	console.log(request.connection.remoteAddress + ' updated their location');
 81 	response.send(players);
 82 };
