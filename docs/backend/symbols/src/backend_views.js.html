 26 /**
 27  * Optionally update the user's location, and fetch 
 28  * the locations of the other players
 29  * 
 30  * @memberOf views
 31  * @name location 
 32  * @param id {Number} The user's id
 33  * @param latitude {Number}	The user's current latitude
 34  * @param longitude {Number} The user's current longitude
 35  * @param accuracy {Number} The accuracy of the location in meters
 36  */
 37 exports.location = function(request, response, method) {	
 38 	if (method === "POST") {
 39 		// Record user's location
 40 		try {
 41 			game_id = request.body.game_id;
 42 			game_data[game_id].players[request.body.user_id] = {
 43 				'latitude': request.body.latitude,
 44 				'longitude': request.body.longitude,
 45 				'accuracy': request.body.accuracy,
 46 				'last_update': new Date(),
 47 				'auth_token': request.body.auth_token
 48 			};
 49 		} catch (e) { }
 50 	}
 51 	
 52 	// Send the players back to the client
 53 	if (game_id && game_data[game_id]) {
 54 		response.send(game_data[game_id]);
 55 	} else {
 56 		response.send({"error": "Invalid game"}, 404);
 57 	}	
 58 };
