  9 /**
 10  * Global memory store for game data
 11  */
 12 var game_data = {};
 13 
 14 // Purge old users
 15 setInterval(function() {
 16     for (var game_iterator in game_data) {
 17         if (game_data.hasOwnProperty(game_iterator)) {
 18         for (var player_iterator in game_data[game_iterator].players) {
 19             if (game_data[game_iterator].players.hasOwnProperty(player_iterator)) {
 20                 // Purge players who haven't updated in over 1 minute
 21                 if (new Date() - game_data[game_iterator].players[player_iterator].last_update >= 1*60*1000) {
 22                     game_data[game_iterator].players[player_iterator].latitude = 0;
 23                     game_data[game_iterator].players[player_iterator].longitude = 0;
 24                     game_data[game_iterator].players[player_iterator].accuracy = 0;
 25                 }
 26                 
 27                 // Reclaim memory of players who haven't updated in 5 minutes
 28                 if (new Date() - game_data[game_iterator].players[player_iterator].last_update >= 5*60*1000) {
 29                     delete game_data[game_iterator].players[player_iterator];
 30                 }
 31             }
 32         }
 33         }
 34     }
 35 }, 1*60*1000);
 36 
 37 // Purge old games
 38 setInterval(function() {
 39     for (var game_iterator in game_data) {
 40             if (game_data.hasOwnProperty(game_iterator)) {
 41             // Delete games that haven't been played on in over 20 minutes
 42             if (new Date() - game_data[game_iterator].last_update >= 20*60*1000) {
 43                 delete game_data[game_iterator];
 44             }
 45         }
 46     }
 47 }, 20*60*1000);

 49 /**
 50  * Update the user's location
 51  * 
 52  * @memberOf views
 53  * @name update_location
 54  */
 55 exports.update_location = function(request, response) {
 56     // Log incoming information for debugging purposes
 57     console.log(request.body);
 58     
 59     // Record user's location
 60 	try {
 61 		game_id = request.body.game_id;
 62 		user_id = request.body.user_id;
 63 		
 64 		if (user_id) {
 65             game_data[game_id].last_update = new Date();
 66             game_data[game_id].players[user_id] = request.body;
 67             game_data[game_id].players[user_id].last_update = new Date();
 68 
 69             //Let the user know the operation was successful
 70             response.send("OK");
 71             return;
 72 		}
 73 	} catch (e) { } 
 74 	
 75 	response.send({"error": "Could not save state"}, 404);
 76 };

 78 /**
 79  * Get the locations of the other players
 80  * 
 81  * @memberOf views
 82  * @name get_location 
 83  */
 84 exports.get_location = function(request, response) {
 85     // Send the players back to the client
 86     game_id = request.query.game_id;
 87     if (game_id && game_data[game_id]) {
 88         response.send(game_data[game_id].players);
 89         //console.log(game_data[game_id]);
 90     } else {
 91         response.send({"error": "Invalid game (" + game_id + ")"}, 404);
 92     }
 93 };

107 /**
108  * Create a new game, and return the id
109  */
110 exports.create_game = function(request, response) {
111     // Generate a new game id
112     game_id = request.body.name;
113     
114     // Create the skeleton of the game
115     game_data[game_id] = {
116         origin: {
117             'latitude': request.body.latitude,
118             'longitude': request.body.longitude
119         },
120         last_update: new Date(),
121         players: {}
122     };
123     
124     // Send confirmation back to client
125     response.send("OK");
126 };
