 17 /**
 18  * Update the user's location, or fetch 
 19  * the locations of the other players

 22  * @name location 
 23  * @param id {Number} The user's id
 24  * @param latitude {Number}	The user's current latitude
 25  * @param longitude {Number} The user's current longitude
 26  * @param accuracy {Number} The accuracy of the location in meters
 27  */
 28 exports.update_location = function(request, response) {
 29 	// Record user's location
 30 	console.log(request.body);
 31 	try {
 32 		game_id = request.body.game_id;
 33 		user_id = request.body.user_id;
 34 		
 35 		if (user_id) {
 36             game_data[game_id].last_update = new Date();
 37             game_data[game_id].players[user_id] = request.body;
 38             game_data[game_id].players[user_id].last_update = new Date();
 39 
 40             //Let the user know the operation was successful
 41             response.send("OK");
 42             return;
 43 		}
 44 	} catch (e) { } 
 45 	
 46 	response.send({"error": "Could not save state"}, 404);
 47 };

 49 exports.get_location = function(request, response) {
 50     // Send the players back to the client
 51     game_id = request.query.game_id;
 52     if (game_id && game_data[game_id]) {
 53         response.send(game_data[game_id].players);
 54         //console.log(game_data[game_id]);
 55     } else {
 56         response.send({"error": "Invalid game (" + game_id + ")"}, 404);
 57     }
 58 };

 72 /**
 73  * Create a new game, and return the id
 74  */
 75 exports.create_game = function(request, response) {
 76     // Generate a new game id
 77     game_id = request.body.name;
 78     
 79     // Create the skeleton of the game
 80     game_data[game_id] = {
 81         origin: {
 82             'latitude': request.body.latitude,
 83             'longitude': request.body.longitude
 84         },
 85         last_update: new Date(),
 86         players: {}
 87     };
 88     
 89     // Send confirmation back to client
 90     response.send("OK");
 91 };
