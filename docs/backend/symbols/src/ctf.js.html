1 /**
  2  * ctf.js
  3  * 
  4  * This is the main application class, and controls
  5  * daemonization, imports, and routing.
  6  * 
  7  * @author Allen Carroll
  8  */
  9 
 10 // Imports
 11 
 12 /**
 13  * The express framework and HTTP middleware
 14  */
 15 var express = require("express");
 16 
 17 /**
 18  * The views which will handle incoming requests
 19  * and return the necessary JSON
 20  * 
 21  * @namespace views
 22  */
 23 var views = require("./backend/views.js");
 24 
 25 /** 
 26  * The server object which controls all routes
 27  * 
 28  * @namespace http
 29  */
 30 var http = express.createServer();
 31 
 32 // Routes
 33 
 34 	//
 35 	// Static routes
 36 	//
 37 
 38 	/**
 39 	 * Index file <br />
 40 	 * <b>url: /</b>
 41 	 * 
 42 	 * @memberOf http
 43 	 * @name root 
 44 	 **/
 45 	http.get('/', function(request, response) {
 46 		response.sendfile("static/index.html");
 47 	});
 48 	
 49 	/**
 50 	 * Static files <br />
 51 	 * <b>url: /*</b>
 52 	 * 
 53 	 * @memberOf http
 54 	 * @name static 
 55 	 **/
 56 	http.use(express.staticProvider('./static/'));
 57 	
 58 	/**
 59 	 * Documentation <br />
 60 	 * <b>url: /*</b>
 61 	 * 
 62 	 * @memberOf http
 63 	 * @name docs 
 64 	 **/
 65 	http.use(express.staticProvider('./docs/'));
 66 	
 67 	//
 68 	// Capture the flag routes
 69 	//	
 70 	
 71 	/**
 72 	 * Calls views.update_location <br />
 73 	 * <b>url: /location</b><br />
 74 	 * methods: PUT
 75 	 * 
 76 	 * @memberOf http
 77 	 * @name location
 78 	 * @link views.update_location 
 79 	 **/
 80 	http.put('/location', views.update_location);
 81 
 82 // Start listening
 83 http.listen(5555);
 84 console.log('CTF server running at http://127.0.0.1:5555/');
