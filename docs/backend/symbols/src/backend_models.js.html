 10 /**
 11 * The settings object
 12 */
 13 var settings = require("../settings.js");
 14 
 15 /**
 16  * The Node.js MongoDB adapter
 17  */
 18 var mongoose = require("mongoose");
 19 var Schema = mongoose.Schema, ObjectId = Schema.ObjectId;
 20 mongoose.connect(settings.mongodb_url);
 21 
 22 /**
 23 * Coordinate object, which holds a GPS coordinate,
 24 * and the accuracy in meters
 25 *
 26 * @memberOf models
 27 * @name Coordinate
 28 */
 29 var Coordinate = new Schema({
 30 	latitude: Number,
 31 	longitude: Number,
 32 	accuracy: Number
 33 });
 34 
 35 /**
 36  * Player object, which holds all information related to the player
 37  *
 38  * @memberOf models
 39  * @name Player
 40  */
 41 
 42 var Player = new Schema({
 43 	id: ObjectId,
 44 	name: { type: String, validate: /[a-z]/ },
 45 	user_id: {type: String }
 46 });
 47 
 48 /**
 49  * Team object, used for proper load balancing of players
 50  * on teams
 51  *
 52  * @memberOf models
 53  * @name Team
 54  */
 55 var Team = new Schema({
 56     players: [Player] 
 57 });
 58 
 59 /**
 60 * Game object, which holds information regarding boundaries
 61 * and game location
 62 *
 63 * @memberOf models
 64 * @name Game
 65 */
 66 
 67 var Game = new Schema({
 68 	id: ObjectId,
 69     teams: [Team],
 70     initial_location: [Coordinate]
 71 });
 72 
 73 // Export
 74 exports.Coordinate = Coordinate;
 75 exports.Player = Player;
 76 exports.Team = Team;
 77 exports.Game = Game;
