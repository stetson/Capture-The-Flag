1 /**

  5  */
  6 var map = {
  7 	/**
  8 	 * Google maps object
  9 	 */
 10 	map: {},

 16 	
 17 	/**
 18 	 * Constructor, called upon page load
 19 	 */
 20 	initialize: function() {
 21 		// Make street level map with no UI controls
 22 		map.options = {
 23 			zoom: 18,
 24             disableDefaultUI: true,
 25 			mapTypeId: google.maps.MapTypeId.HYBRID
 26 		};
 27 		
 28 		// Check for geolocation support
 29 		if (navigator.geolocation) {
 30 			// Generate map and render on page
 31 			map.map = new google.maps.Map(document.getElementById("map_canvas"), map.options);
 32 			
 33 			// Center map
 34 			model.centerMap();
 35             
 36             // Get user credentials
 37             // FIXME - This should be in the callback from joining a game
 38 			// model.login();
 39             model.watchLocation();
 40 		} else {
 41 			$("#loading").html("Your device does not appear to support HTML5 geolocation");
 42 		}
 43 	}
 44 };

 93 	/**
 94 	 * Center map after the first load
 95 	 */
 96 	centerMap: function() {
 97 		navigator.geolocation.getCurrentPosition(function(position) {
 98 			map.map.setCenter(new google.maps.LatLng(position.coords.latitude, position.coords.longitude));
 99 		}, function() {}, {
100 			maximumAge: 3000,
101 			enableHighAccuracy: true
102 		});
103 	},
104 	
105 	/**
106 	 * Center the user on their own location, and set up location listener
107 	 */
108 	watchLocation: function() {
109 		try {
110 			
111 			// Trigger update when position moves
112 			navigator.geolocation.watchPosition(
113 				// Success
114 				function(position) {
115 					model.updateLocation(position);
116 				}, 
117 				
118 				// Failure
119 				function() {
120 					$("#loading").html("Your location was not available. Please ensure that you have given permissions for geolocation.");
121 				},
122 				
123 				// Options
124 				{
125 					maximumAge: 1000,
126 					enableHighAccuracy: true
127 				}
128 			);
129 		
130 		} catch (e) {
131 			$("#loading").html("Browser did not report location.");
132 		}
133 	},
134 	
135 	/**
136 	 * Server call which updates your current location and gets
137 	 * the locations of all the other players
138 	 */
139 	updateLocation: function(position) {
140 		if (position.coords.accuracy < 30) {
141 			$("#loading").fadeOut('slow');
142 			$.ajax({
143 		        url: '/location/',
144 		        type: 'POST',
145 		        data: {
146 					user_id: model.auth_token,
147 		            latitude: position.coords.latitude,
148 		            longitude: position.coords.longitude,
149 		            accuracy: position.coords.accuracy
150 		        },
151 		        dataType: 'json',
152 		        success: function(data) {
153 		        	// Don't update if empty response from the server
154 		        	if (!data)
155 		        		return;
156 		        	
157 		        	model.players = data;
158 		        	
159 		        	// Update the locations of each player
160 		        	$.each(data, function(player_iterator, player) {
161 		        		if (model.player_markers[player_iterator] == undefined) {
162 		        			icon = player_iterator == model.auth_token ? "/css/images/star.png" : "/css/images/person.png";
163 		        			model.player_markers[player_iterator] = new google.maps.Marker({
164 								position: new google.maps.LatLng(player.latitude, player.longitude),
165 								map: map.map,
166 								title: "Player " + player_iterator,
167 								icon: icon
168 							});
169 		        		} else {
170 		        			model.player_markers[player_iterator].position = new google.maps.LatLng(player.latitude, player.longitude);
171 		        		}
172 		        	});
173 		        }
174 		    });
175 		} else {
176 			$("#loading").show();
177 			$("#loading").html("Accuracy is still not acceptable (" + position.coords.accuracy + "m). <br />If this error persists, please ensure that your GPS radio is on.
