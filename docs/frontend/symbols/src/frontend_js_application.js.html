1 var model;
  2 var map;
  3 
  4 /**
  5  * Map object which handles geolocation and mapping
  6  * 
  7  * @namespace map
  8  */
  9 map = {
 10 	/**
 11 	 * Google maps object
 12 	 */
 13 	map: {},
 14 	
 15 	/**
 16 	 * Map options
 17 	 */
 18 	options: {},
 19 	
 20 	/**
 21 	 * Constructor, called upon page load
 22 	 */
 23 	initialize: function() {
 24 		// Make street level map with no UI controls
 25 		map.options = {
 26 			zoom: 18,
 27             disableDefaultUI: true,
 28 			mapTypeId: google.maps.MapTypeId.HYBRID
 29 		};
 30 		
 31 		// Check for geolocation support
 32 		if (navigator.geolocation) {
 33 			// Generate map and render on page
 34 			map.map = new google.maps.Map(document.getElementById("map_canvas"), map.options);
 35 			
 36 			// Center map
 37 			model.centerMap();
 38 			
 39 			// Watch the current location (lock onto GPS while user is reading the front page)
 40             model.watchLocation();
 41 		} else {
 42 			model.error("Your device does not appear to support HTML5 geolocation");
 43 		}
 44 	}
 45 };
 46 
 47 /**
 48  * Model which handles server calls
 49  * 
 50  * @namespace model
 51  */
 52 model = {
 53 		
 54 	/**
 55 	 * Facebook App ID
 56 	 */
 57 	app_id: 151829711542674,
 58 		
 59 	/**
 60 	 * Facebook Connect authentication token
 61 	 */
 62 	auth_token: "",
 63 	
 64 	/**
 65 	 * Facebook id
 66 	 */
 67 	user_id: "",
 68 	
 69 	/**
 70 	 * Conditions have been met for gameplay
 71 	 */
 72 	game_in_progress: false,
 73 	
 74 	/**
 75 	 * Array which holds the player objects
 76 	 */
 77 	players: {},
 78 	
 79 	/**
 80 	 * Array which holds the player markers
 81 	 */
 82 	player_markers: {},
 83 	
 84 	/**
 85 	 * Initialize Facebook Connect login
 86 	 */
 87 	login: function() {
 88 		FB.init({
 89 			appId  : model.app_id,
 90 			status : true, // check login status
 91 			cookie : true, // enable cookies to allow the server to access the session
 92 			xfbml  : true,  // parse XFBML
 93 			popup: false
 94 		});
 95 		
 96 		// Hook authentication into Facebook Connect
 97 		FB.getLoginStatus(function(response) {
 98 			  if (response.session) {
 99 				  // user was already logged in
100 				  model.auth_token = response.session.access_token;
101 				  model.user_id = response.session.uid;
102 				  model.login_successful();
103 			  } else {
104 				  // let user log in whenever they darn well please
105 			  }
106 		});
107 	},
108 	
109 	/**
110 	 * Log in as guest instead of using Facebook Connect
111 	 */
112 	guest_login: function() {
113 		model.auth_token = "guest";
114 		model.user_id = 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
115 		    var r = Math.random()*16|0, v = c == 'x' ? r : (r&0x3|0x8);
116 		    return v.toString(16);
117 		}).toUpperCase();
118 
119 		model.login_successful();
120 		return false;
121 	},
122 	
123 	/**
124 	 * Called after session is successfully retrieved
125 	 */
126 	login_successful: function() {
127 		try {
128 			window.location.hash = "#play";
129 			window.location.search = "";
130 		} catch (e) { }
131 
132 		$("#fb-root").hide();
133 		$("#loading").fadeOut("slow");
134 	},
135 	
136 	/**
137 	 * Something isn't right
138 	 */
139 	error: function(message) {
140 		$("#loading").fadeIn();
141 		$(".error").show();
142 		$(".error").html(message);	
143 	},
144 	
145 	/**
146 	 * Center map after the first load
147 	 */
148 	centerMap: function() {
149 		navigator.geolocation.getCurrentPosition(function(position) {
150 			map.map.setCenter(new google.maps.LatLng(position.coords.latitude, position.coords.longitude));
151 		}, function() {}, {
152 			maximumAge: 500,
153 			enableHighAccuracy: true
154 		});
155 	},
156 	
157 	/**
158 	 * Center the user on their own location, and set up location listener
159 	 */
160 	watchLocation: function() {
161 		try {
162 			
163 			// Trigger update when position moves
164 			navigator.geolocation.watchPosition(
165 				// Success
166 				function(position) {
167 					model.updateLocation(position);
168 				}, 
169 				
170 				// Failure
171 				function() {
172 					model.error("Your location was not available. Please ensure that you have given permissions for geolocation.");
173 				},
174 				
175 				// Options
176 				{
177 					maximumAge: 1000,
178 					enableHighAccuracy: true
179 				}
180 			);
181 		
182 		} catch (e) {
183 			model.error("Browser did not report location.");
184 		}
185 	},
186 	
187 	/**
188 	 * Server call which updates your current location and gets
189 	 * the locations of all the other players
190 	 */
191 	updateLocation: function(position) {
192 		// Update your location, regardless of whether it's in strict accuracy requirements
193 		model.player_markers[model.user_id].position = new google.maps.LatLng(position.coords.latitude, position.coords.longitude);
194 		
195 		// Update the server if strict requirements have been met
196 		if (position.coords.accuracy <= 30 && model.auth_token !== "") {
197 			$("#loading").fadeOut('slow');
198 			$.ajax({
199 		        url: '/location/',
200 		        type: 'POST',
201 		        data: {
202 					user_id: model.user_id,
203 					auth_token: model.auth_token,
204 		            latitude: position.coords.latitude,
205 		            longitude: position.coords.longitude,
206 		            accuracy: position.coords.accuracy
207 		        },
208 		        dataType: 'json',
209 		        success: function(data) {
210                     // Don't update if empty response from the server
211 		        	if (!data) {
212 		        		return;
213 		        	}
214 
215 		        	model.players = data;
216 
217 		        	// Update the locations of each player
218 		        	$.each(data, function(player_iterator, player) {
219 		        		if (model.player_markers[player_iterator] === undefined) {
220 		        			icon = player_iterator == model.user_id ? "/css/images/star.png" : "/css/images/person.png";
221 		        			model.player_markers[player_iterator] = new google.maps.Marker({
222 								position: new google.maps.LatLng(player.latitude, player.longitude),
223 								map: map.map,
224 								title: "Player " + player_iterator,
225 								icon: icon
226 							});
227 		        		} else {
228 		        			model.player_markers[player_iterator].position = new google.maps.LatLng(player.latitude, player.longitude);
229 		        		}
230 		        	});
231 		        }
232 		    });
233 		}
234 	}
235 };
236 
237 // Start application
238 $(document).ready(function() {
239 	$(".error").hide();
240 	$("#guest-login").click(model.guest_login);
241 	
242 	// Asynchronously load Facebook Connect js, then login
243 	$.getScript("http://connect.facebook.net/en_US/all.js", function() {
244 		model.login();
245 	});
246 	
247 	// In the background, lock onto the user's location
248 	map.initialize();
249 });
