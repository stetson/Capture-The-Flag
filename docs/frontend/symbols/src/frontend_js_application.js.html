 32             
 33             // Get user credentials
 34             // FIXME - This should be in model.login();
 35             model.watchLocation();

 37 			// Get rid of loading message once map is loaded
 38 			// TODO - this will be replaced by login window
 39             google.maps.event.addListener(map.map, 'tilesloaded', function() {
 40                 $("#loading").fadeOut('slow');
 41             });
 42 		} else {
 43 			alert("Your device does not appear to support HTML5 geolocation");
 44 		}
 45 	}

104 	/**
105 	 * Center the user on their own location, and set up location listener
106 	 */
107 	watchLocation: function() {
108 		try {
109 			
110 			// Trigger update when position moves
111 			navigator.geolocation.watchPosition(
112 				// Success
113 				function(position) {
114 					model.updateLocation(position);
115 				}, 
116 				
117 				// Failure
118 				function() {
119 					alert("Your device has not given permission for HTML5 geolocation");
120 				},
121 				
122 				// Options
123 				{
124 					maximumAge: 3000,
125 					enableHighAccuracy: true
126 				}
127 			);
128 		
129 		} catch (e) {
130 			alert("Browser did not report location.");
131 		}
132 		
133 		setInterval(function() {
134 			// Trigger update on an interval
135 			navigator.geolocation.getCurrentPosition(
136 				// Success
137 				function(position) {
138 					model.updateLocation(position);
139 				}, 
140 				
141 				// Failure
142 				function() {
143 					alert("Your device has not given permission for HTML5 geolocation");
144 				},
145 				
146 				// Options
147 				{
148 					maximumAge: 3000,
149 					enableHighAccuracy: true
150 				}
151 			);
152 		}, 5000);
153 	},
154 	
155 	/**
156 	 * Server call which updates your current location and gets
157 	 * the locations of all the other players
158 	 */
159 	updateLocation: function(position) {
160 		//console.debug(position);
161 		// TODO - check for accuracy
162 		//if (position.coords.accuracy < 30) {
163 	    	// Center map on your new coordinates
164 			map.map.setCenter(new google.maps.LatLng(position.coords.latitude, position.coords.longitude));
165 
166 			$.ajax({
167 		        url: '/location/',
168 		        type: 'PUT',
169 		        data: {
170 					user_id: model.auth_token,
171 		            latitude: position.coords.latitude,
172 		            longitude: position.coords.longitude,
173 		            accuracy: position.coords.accuracy
174 		        },
175 		        dataType: 'json',
176 		        success: function(data) {
177 		        	// Don't update if empty response from the server
178 		        	if (!data)
179 		        		return;
180 		        	
181 		        	model.players = data;
182 		        	
183 		        	// Update the locations of each player
184 		        	$.each(data, function(player_iterator, player) {
185 		        		if (model.player_markers[player_iterator] == undefined) {
186 		        			model.player_markers[player_iterator] = new google.maps.Marker({
187 								position: new google.maps.LatLng(player.latitude, player.longitude),
188 								map: map.map,
189 								title: "Player " + player_iterator
190 							});
191 		        		} else {
192 		        			model.player_markers[player_iterator].position = new google.maps.LatLng(player.latitude, player.longitude);
193 		        		}
194 		        	});
195 		        }
196 		    });
197 		//} else {
198 		//	$("#loading").html("Accuracy is still not acceptable (" + position.coords.accuracy + "). <br />If this persists, please ensure that your GPS radio is on.");
199 		//}
200 	}
