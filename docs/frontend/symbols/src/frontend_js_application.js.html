1 /**
  2  * Map object which handles geolocation and mapping
  3  * 
  4  * @namespace map
  5  */
  6 var map = {
  7 	/**
  8 	 * Google maps object
  9 	 */
 10 	map: {},
 11 	
 12 	/**
 13 	 * Map options
 14 	 */
 15 	options: {},
 16 	
 17 	/**
 18 	 * Constructor, called upon page load
 19 	 */
 20 	initialize: function() {
 21 		// Make street level map with no UI controls
 22 		map.options = {
 23 			zoom: 18,
 24             disableDefaultUI: true,
 25 			mapTypeId: google.maps.MapTypeId.HYBRID
 26 		};
 27 		
 28 		// Check for geolocation support
 29 		if (navigator.geolocation) {
 30 			// Generate map and render on page
 31 			map.map = new google.maps.Map(document.getElementById("map_canvas"), map.options);
 32 			
 33 			// Center map
 34 			model.centerMap();
 35             
 36             // Get user credentials
 37             // FIXME - This should be in the callback from joining a game
 38 			// model.login();
 39             model.watchLocation();
 40 		} else {
 41 			$("#loading").html("Your device does not appear to support HTML5 geolocation");
 42 		}
 43 	}
 44 };
 45 
 46 /**
 47  * Model which handles server calls
 48  * 
 49  * @namespace model
 50  */
 51 var model = {
 52 		
 53 	/**
 54 	 * Facebook Connect authentication token
 55 	 * FIXME - this is just a random UUID at the moment
 56 	 */
 57 	auth_token: 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
 58 	    var r = Math.random()*16|0, v = c == 'x' ? r : (r&0x3|0x8);
 59 	    return v.toString(16);
 60 	}).toUpperCase(),
 61 	
 62 	/**
 63 	 * Conditions have been met for gameplay
 64 	 */
 65 	game_in_progress: false,
 66 	
 67 	/**
 68 	 * Array which holds the player objects
 69 	 */
 70 	players: {},
 71 	
 72 	/**
 73 	 * Array which holds the player markers
 74 	 */
 75 	player_markers: {},
 76 	
 77 	// Get a session for the current user
 78 	/*
 79 	login: function() {
 80 		// Do Facebook Connect magic here, and assign auth token to model.auth_token
 81 	},
 82 	*/
 83 	
 84 	/**
 85 	 * Center map after the first load
 86 	 */
 87 	centerMap: function() {
 88 		navigator.geolocation.getCurrentPosition(function(position) {
 89 			map.map.setCenter(new google.maps.LatLng(position.coords.latitude, position.coords.longitude));
 90 		}, function() {}, {
 91 			maximumAge: 3000,
 92 			enableHighAccuracy: true
 93 		});
 94 	},
 95 	
 96 	/**
 97 	 * Center the user on their own location, and set up location listener
 98 	 */
 99 	watchLocation: function() {
100 		try {
101 			
102 			// Trigger update when position moves
103 			navigator.geolocation.watchPosition(
104 				// Success
105 				function(position) {
106 					model.updateLocation(position);
107 				}, 
108 				
109 				// Failure
110 				function() {
111 					$("#loading").html("Your location was not available. Please ensure that you have given permissions for geolocation.");
112 				},
113 				
114 				// Options
115 				{
116 					maximumAge: 1000,
117 					enableHighAccuracy: true
118 				}
119 			);
120 		
121 		} catch (e) {
122 			$("#loading").html("Browser did not report location.");
123 		}
124 	},
125 	
126 	/**
127 	 * Server call which updates your current location and gets
128 	 * the locations of all the other players
129 	 */
130 	updateLocation: function(position) {
131 		if (position.coords.accuracy < 30) {
132 			$("#loading").fadeOut('slow');
133 			$.ajax({
134 		        url: '/location/',
135 		        type: 'POST',
136 		        data: {
137 					user_id: model.auth_token,
138 		            latitude: position.coords.latitude,
139 		            longitude: position.coords.longitude,
140 		            accuracy: position.coords.accuracy
141 		        },
142 		        dataType: 'json',
143 		        success: function(data) {
144 		        	// Don't update if empty response from the server
145 		        	if (!data)
146 		        		return;
147 		        	
148 		        	model.players = data;
149 		        	
150 		        	// Update the locations of each player
151 		        	$.each(data, function(player_iterator, player) {
152 		        		if (model.player_markers[player_iterator] == undefined) {
153 		        			icon = player_iterator == model.auth_token ? "/css/images/star.png" : "/css/images/person.png";
154 		        			model.player_markers[player_iterator] = new google.maps.Marker({
155 								position: new google.maps.LatLng(player.latitude, player.longitude),
156 								map: map.map,
157 								title: "Player " + player_iterator,
158 								icon: icon
159 							});
160 		        		} else {
161 		        			model.player_markers[player_iterator].position = new google.maps.LatLng(player.latitude, player.longitude);
162 		        		}
163 		        	});
164 		        }
165 		    });
166 		} else {
167 			$("#loading").show();
168 			$("#loading").html("Accuracy is still not acceptable (" + position.coords.accuracy + "m). <br />If this error persists, please ensure that your GPS radio is on.");
169 		}
170 	}
171 };
172 
173 // Start application
174 $(document).ready(function() {
175 	map.initialize();
176 });
