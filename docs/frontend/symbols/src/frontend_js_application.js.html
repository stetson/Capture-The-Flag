 17 	/**
 18 	 * Constructor, called upon page load
 19 	 */
 20 	initialize: function() {
 21 		// Make street level map with no UI controls
 22 		map.options = {
 23 			zoom: 18,
 24             disableDefaultUI: true,
 25 			mapTypeId: google.maps.MapTypeId.HYBRID
 26 		};
 27 		
 28 		// Check for geolocation support
 29 		if (navigator.geolocation) {
 30 			// Generate map and render on page
 31 			map.map = new google.maps.Map(document.getElementById("map_canvas"), map.options);
 32             
 33             // Get user credentials
 34             // FIXME - This should be in model.login();
 35             model.watchLocation();
 36             
 37 			// Get rid of loading message once map is loaded
 38 			// TODO - this will be replaced by login window
 39             google.maps.event.addListener(map.map, 'tilesloaded', function() {
 40                 $("#loading").fadeOut('slow');
 41             });
 42 		} else {
 43 			alert("Your device does not appear to support HTML5 geolocation");
 44 		}
 45 	}
 46 };

 94 	
 95 	/**
 96 	 * Center the user on their own location, and set up location listener
 97 	 */
 98 	watchLocation: function() {
 99 		try {
100 			
101 			// Trigger update when position moves
102 			navigator.geolocation.watchPosition(
103 				// Success
104 				function(position) {
105 					model.updateLocation(position);
106 				}, 
107 				
108 				// Failure
109 				function() {
110 					alert("Your device has not given permission for HTML5 geolocation");
111 				},
112 				
113 				// Options
114 				{
115 					maximumAge: 3000,
116 					enableHighAccuracy: true
117 				}
118 			);
119 		
120 		} catch (e) {
121 			alert("Browser did not report location.");
122 		}
123 	},
124 	
125 	/**
126 	 * Server call which updates your current location and gets
127 	 * the locations of all the other players
128 	 */
129 	updateLocation: function(position) {
130 		//console.debug(position);
131 		// TODO - check for accuracy
132 		//if (position.coords.accuracy < 30) {
133 	    	// Center map on your new coordinates
134 			map.map.setCenter(new google.maps.LatLng(position.coords.latitude, position.coords.longitude));
135 
136 			$.ajax({
137 		        url: '/api/location/',
138 		        type: 'PUT',
139 		        data: {
140 					uuid: model.uuid,
141 		            latitude: position.coords.latitude,
142 		            longitude: position.coords.longitude
143 		        },
144 		        dataType: 'json',
145 		        success: function(data) {
146 		        	model.players = data;
147 		        	
148 		        	// Update the locations of each player
149 		        	$.each(data, function(player_iterator, player) {
150 		        		if (model.player_markers[player.id] == undefined) {
151 		        			model.player_markers[player.id] = new google.maps.Marker({
152 								position: new google.maps.LatLng(player.latitude, player.longitude),
153 								map: map.map,
154 								title: "Player " + player.id
155 							});
156 		        		} else {
157 		        			model.player_markers[player.id].position = new google.maps.LatLng(player.latitude, player.longitude);
158 		        		}
159 		        	});
160 		        }
161 		    });
162 		//} else {
163 		//	$("#loading").html("Accuracy is still not acceptable (" + position.coords.accuracy + "). <br />If this persists, please ensure that your GPS radio is on.");
164 		//}
165 	}
